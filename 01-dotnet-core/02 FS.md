## SOLID Principles with C# Examples

### 1. Single Responsibility Principle (SRP)

> *A class should have only one reason to change.*

**Violation Example:**
```csharp
public class OrderProcessor
{
    public void ProcessOrder(Order order)
    {
        // 1. Validate order
        if (order.Items.Count == 0) throw new Exception("No items");
        
        // 2. Calculate total
        decimal total = order.Items.Sum(i => i.Price * i.Quantity);
        
        // 3. Save to database
        using (var conn = new SqlConnection("..."))
        {
            // Save order...
        }
        
        // 4. Send confirmation email
        SmtpClient client = new SmtpClient();
        client.Send(new MailMessage("shop@test.com", order.CustomerEmail, "Order placed", "Your order is confirmed"));
    }
}
```

**Refactored (SRP Applied):**
```csharp
public class OrderValidator
{
    public void Validate(Order order) { /* validation logic */ }
}

public class OrderCalculator
{
    public decimal CalculateTotal(Order order) => order.Items.Sum(i => i.Price * i.Quantity);
}

public class OrderRepository
{
    public void Save(Order order) { /* database persistence */ }
}

public class EmailService
{
    public void SendOrderConfirmation(string email, string message) { /* email sending */ }
}

public class OrderProcessor
{
    private readonly OrderValidator _validator;
    private readonly OrderCalculator _calculator;
    private readonly OrderRepository _repository;
    private readonly EmailService _emailService;

    public OrderProcessor(OrderValidator validator, OrderCalculator calculator,
                          OrderRepository repository, EmailService emailService)
    {
        _validator = validator;
        _calculator = calculator;
        _repository = repository;
        _emailService = emailService;
    }

    public void ProcessOrder(Order order)
    {
        _validator.Validate(order);
        order.Total = _calculator.CalculateTotal(order);
        _repository.Save(order);
        _emailService.SendOrderConfirmation(order.CustomerEmail, "Order placed");
    }
}
```

### 2. Open/Closed Principle (OCP)

> *Software entities should be open for extension, but closed for modification.*

**Violation:**
```csharp
public class DiscountCalculator
{
    public decimal CalculateDiscount(Customer customer, decimal amount)
    {
        if (customer.Type == CustomerType.Regular)
            return amount * 0.05m;
        else if (customer.Type == CustomerType.Premium)
            return amount * 0.10m;
        else if (customer.Type == CustomerType.VIP)
            return amount * 0.20m;
        return 0;
    }
}
```

**Refactored (Strategy Pattern + OCP):**
```csharp
public interface IDiscountStrategy
{
    decimal Calculate(decimal amount);
}

public class RegularDiscount : IDiscountStrategy
{
    public decimal Calculate(decimal amount) => amount * 0.05m;
}

public class PremiumDiscount : IDiscountStrategy
{
    public decimal Calculate(decimal amount) => amount * 0.10m;
}

public class VipDiscount : IDiscountStrategy
{
    public decimal Calculate(decimal amount) => amount * 0.20m;
}

public class DiscountCalculator
{
    private readonly IDictionary<CustomerType, IDiscountStrategy> _strategies;

    public DiscountCalculator()
    {
        _strategies = new Dictionary<CustomerType, IDiscountStrategy>
        {
            [CustomerType.Regular] = new RegularDiscount(),
            [CustomerType.Premium] = new PremiumDiscount(),
            [CustomerType.VIP] = new VipDiscount()
        };
    }

    public decimal CalculateDiscount(CustomerType type, decimal amount) =>
        _strategies.TryGetValue(type, out var strategy) ? strategy.Calculate(amount) : 0;
}
```

Now adding a new discount type requires only creating a new class and registering it – no changes to `DiscountCalculator`.

### 3. Liskov Substitution Principle (LSP)

> *Derived classes must be substitutable for their base classes without altering correctness.*

**Violation – Rectangle and Square:**
```csharp
public class Rectangle
{
    public virtual int Width { get; set; }
    public virtual int Height { get; set; }
    public int Area => Width * Height;
}

public class Square : Rectangle
{
    public override int Width
    {
        set { base.Width = base.Height = value; }
    }
    public override int Height
    {
        set { base.Width = base.Height = value; }
    }
}

// Client code expecting Rectangle behaviour fails:
Rectangle rect = new Square();
rect.Width = 5;
rect.Height = 10;   // In Square, this sets both to 10
Console.WriteLine(rect.Area); // Expect 50, gets 100
```

**Fix – avoid inheritance for incompatible abstractions:**
```csharp
public interface IShape
{
    int Area { get; }
}

public class Rectangle : IShape
{
    public int Width { get; set; }
    public int Height { get; set; }
    public int Area => Width * Height;
}

public class Square : IShape
{
    public int Side { get; set; }
    public int Area => Side * Side;
}
```

### 5. Dependency Inversion Principle (DIP)

> *High‑level modules should not depend on low‑level modules. Both should depend on abstractions.*

**Violation:**
```csharp
public class EmailNotifier
{
    public void Send(string message) { /* SMTP code */ }
}

public class OrderService
{
    private EmailNotifier _notifier = new EmailNotifier();
    public void CompleteOrder() { _notifier.Send("Order completed"); }
}
```

**Refactored (DIP applied):**
```csharp
public interface INotifier
{
    void Send(string message);
}

public class EmailNotifier : INotifier { /* ... */ }
public class SmsNotifier : INotifier { /* ... */ }

public class OrderService
{
    private readonly INotifier _notifier;
    public OrderService(INotifier notifier) => _notifier = notifier;
    public void CompleteOrder() => _notifier.Send("Order completed");
}
```

**Dependency Injection** is the natural implementation of DIP in .NET Core.

### 9. Specification Pattern

```csharp
public interface ISpecification<T>
{
    Expression<Func<T, bool>> Criteria { get; }
    List<Expression<Func<T, object>>> Includes { get; }
    Expression<Func<T, object>> OrderBy { get; }
    // ...
}

public class ProductsOnSaleSpecification : ISpecification<Product>
{
    public Expression<Func<Product, bool>> Criteria => p => p.Discount > 0;
    public List<Expression<Func<Product, object>>> Includes => new();
    public Expression<Func<T, object>> OrderBy => p => p.Name;
}

public class SpecificationEvaluator<T> where T : class
{
    public static IQueryable<T> GetQuery(IQueryable<T> inputQuery, ISpecification<T> spec)
    {
        var query = inputQuery;
        if (spec.Criteria != null)
            query = query.Where(spec.Criteria);
        
        query = spec.Includes.Aggregate(query, (current, include) => current.Include(include));
        
        if (spec.OrderBy != null)
            query = query.OrderBy(spec.OrderBy);
        
        return query;
    }
}
```

### 1. Implementing a Thread‑Safe Generic Cache with Expiration

```csharp
public class TimedCache<TKey, TValue> where TKey : notnull
{
    private readonly ConcurrentDictionary<TKey, CacheEntry> _cache = new();
    private readonly Timer _cleanupTimer;

    public TimedCache(TimeSpan cleanupInterval)
    {
        _cleanupTimer = new Timer(CleanupExpired, null, cleanupInterval, cleanupInterval);
    }

    public void Add(TKey key, TValue value, TimeSpan ttl)
    {
        _cache[key] = new CacheEntry(value, DateTime.UtcNow.Add(ttl));
    }

    public bool TryGet(TKey key, out TValue value)
    {
        if (_cache.TryGetValue(key, out var entry) && entry.ExpiresAt > DateTime.UtcNow)
        {
            value = entry.Value;
            return true;
        }
        value = default!;
        return false;
    }

    private void CleanupExpired(object state)
    {
        var now = DateTime.UtcNow;
        foreach (var kvp in _cache.Where(kvp => kvp.Value.ExpiresAt <= now))
            _cache.TryRemove(kvp.Key, out _);
    }

    private record CacheEntry(TValue Value, DateTime ExpiresAt);
}
```

### 2. Async/Await Pitfall: Avoiding Deadlocks with `ConfigureAwait(false)`

### 3. Implementing a Producer‑Consumer Queue with `Channel<T>`

```csharp
public class BackgroundTaskQueue
{
    private readonly Channel<Func<CancellationToken, ValueTask>> _queue;

    public BackgroundTaskQueue(int capacity = 100)
    {
        var options = new BoundedChannelOptions(capacity) { FullMode = BoundedChannelFullMode.Wait };
        _queue = Channel.CreateBounded<Func<CancellationToken, ValueTask>>(options);
    }

    public async ValueTask QueueBackgroundWorkItemAsync(Func<CancellationToken, ValueTask> workItem)
    {
        await _queue.Writer.WriteAsync(workItem);
    }

    public async ValueTask<Func<CancellationToken, ValueTask>> DequeueAsync(CancellationToken cancellationToken)
    {
        return await _queue.Reader.ReadAsync(cancellationToken);
    }
}

// Hosted Service consumer:
public class QueuedHostedService : BackgroundService
{
    private readonly BackgroundTaskQueue _queue;
    public QueuedHostedService(BackgroundTaskQueue queue) => _queue = queue;

    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        while (!stoppingToken.IsCancellationRequested)
        {
            var workItem = await _queue.DequeueAsync(stoppingToken);
            await workItem(stoppingToken);
        }
    }
}
```

```csharp
public async Task ProcessDataStreamAsync()
{
    var channel = Channel.CreateUnbounded<string>();

    var producer = Task.Run(async () =>
    {
        for (int i = 0; i < 1000; i++)
        {

        }
```

```csharp
public static IQueryable<T> FilterByProperty<T>(IQueryable<T> source, string propertyName, object value)
{
    var parameter = Expression.Parameter(typeof(T), "x");
    var property = Expression.Property(parameter, propertyName);
    var constant = Expression.Constant(value);
    var equality = Expression.Equal(property, constant);
    var lambda = Expression.Lambda<Func<T, bool>>(equality, parameter);

}
```

```csharp
var summary = db.Orders
    .GroupBy(o => new { o.CustomerId, Year = o.OrderDate.Year })
    .Select(g => new
    {
        g.Key.CustomerId,
        g.Key.Year,
        TotalAmount = g.Sum(o => o.Total),
        OrderCount = g.Count()
    });
```
